def preprocess_image(image: np.ndarray) -> np.ndarray:
    """
    image: (H, W, 3) uint8 array
    returns: (448, 448, 3) float array
    """
    from PIL import Image

    # è½¬ä¸º PIL å›¾åƒ
    pil_image = Image.fromarray(image)

    # è°ƒæ•´å¤§å°
    pil_image = pil_image.resize((448, 448))
